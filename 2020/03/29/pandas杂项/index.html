常用代码片段

%matplotlib inline
import matplotlib as mpl
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
import seaborn as sns
sns.set()

plt.rcParams['font.sans-serif'] = ['Arial Unicode MS']
mpl.rcParams["axes.unicode_minus"] = False
